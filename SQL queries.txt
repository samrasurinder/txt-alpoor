UNION
SELECT 'CRNT_PRSN', COUNT(*) FROM DW.CRNT_PRSN
UNION
SELECT 'CRNT_POSTN',COUNT(*) FROM DW.CRNT_POSTN
UNION
SELECT 'POSITION', COUNT(*) FROM DW.POSITION
UNION
SELECT 'PERSON_PENDING', COUNT(*) FROM dw.PERSON_PENDING
UNION
SELECT 'BUREAU COUNT FROM DEPT TABLE', COUNT(*) 
FROM   DW.DEPARTMENT WHERE BUREAU IS NOT NULL
AND BUREAU_ORG_CD IS NOT NULL AND BUREAU_DESCR IS NOT NULL
