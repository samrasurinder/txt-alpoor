
Step 1) Extract schema and ACLs using following SQLs

select PRINCIPAL,acl from dba_network_acl_privileges where principal in (SELECT USERNAME FROM DBA_USERS WHERE PROFILE='HR_PROFILE') ORDER BY 1;


Step2 ) Delete privilege for each pricipal and acl extracted from step 1 as follows:

BEGIN
  DBMS_NETWORK_ACL_ADMIN.DELETE_PRIVILEGE(
        acl         => 'NETWORK_ACL_CF57C224C02E03E6E0532A69440A39E0',
        principal   => 'IDB');
end;


Step3) Re-add privileges after task is complete as follows for each principal extracted from step 1

BEGIN
DBMS_NETWORK_ACL_ADMIN.ADD_PRIVILEGE(
acl => 'NETWORK_ACL_CF57C224C02E03E6E0532A69440A39E0',
principal => 'IDB',  --if you want user like 'HR' 
is_grant => true,
privilege => 'connect');
COMMIT;
END;
